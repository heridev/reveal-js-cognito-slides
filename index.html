Then you can implement something else... 

						
							// The "I handle refreshes like a pro" approach
								import {
									fetchAuthSession
								} from 'aws-amplify/auth';

								class TokenManager {
									private refreshPromise: Promise | null = null;

									async getValidToken() {
										try {
											const {
												tokens,
												hasValidToken
											} = await fetchAuthSession({
												// Let's be gentle with our servers
												forceRefresh: false
											});

											if (!hasValidToken) {
												// Time for a refresh!
												return this.refreshSessionIfNeeded();
											}

											return tokens!.accessToken;
										} catch (error) {
											console.error("Token machine broke");
											throw error;
										}
									}

									private async refreshSessionIfNeeded() {
										// Only refresh once,
										// no matter how many requests are waiting
										if (!this.refreshPromise) {
											this.refreshPromise = (
												async () => {
												  try {
														const { tokens } = await fetchAuthSession({
															// Now we mean business
															forceRefresh: true
														});
														return tokens!.accessToken;
													} finally {
														this.refreshPromise = null;
												}
											})();
										}

										return this.refreshPromise;
									}
								}

								// Usage in your API client
								const tokenManager = new TokenManager();

								axios
								  .interceptors
								  .request
									.use(
										async (config) => {
											const token = await tokenManager.getValidToken();
											config.headers.Authorization
											  = `Bearer ${token}`;
											return config;
										});
